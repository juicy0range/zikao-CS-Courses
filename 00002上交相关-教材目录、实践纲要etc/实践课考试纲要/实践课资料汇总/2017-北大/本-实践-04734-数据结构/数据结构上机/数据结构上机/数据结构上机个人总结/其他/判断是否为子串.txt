/***************************************************************\
8．编一C程序，它能读入6个字符串（每个串长度都不大于80，可能含空格字符），
依次判断前5个串是否为第6个串的子串，若是则输出YES；否则输出NO。

abcdefg 7
ade  3
\***************************************************************/

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream.h>

int substr(char *str1,char *str2)
{//功能:该函数判断字符串str2 是否为 str1的子串
	int n = strlen(str1);
	int m = strlen(str2);
	int j,k;
	for(int i = 0;i<=n-m;i++)	//注意n-m 
	{
		j = i;
		k = 0;
		while (str2[k] == str1[j] && k<m) {
			k++;j++;
		}
		if(k == m)	//说明str2已经遍历完,是str1的字串
		return 1;
	}
	return 0;
}


void main()
{

	char *str[6];
	int i = 0;
	for(i = 0 ;i<6;i++)
	{
		cout<<"请输入字符串"<<i+1<<endl;
		str[i] = new char[80];
		cin>>str[i];
	}

	for(i = 0 ;i < 5;i++)
	{
		if(substr(str[5],str[i]))
		{
		
			cout<< i+1 <<" is Yes "<<endl;
		}
		else
		{
			cout<< i+1 <<" is NO "<<endl;
		}
	}

	for(i = 0 ;i<6;i++)
	{
		delete[] (str[i]);		//
	}
	
	getchar();
}
